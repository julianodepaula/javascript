Fixar casas decimais
    variavel.toFixed(quatas casas)
    var n1 = 1545.5
    Ex: n1.toFixed(2)
        retorna '1545.50'
    se quiser com virgula
        n1.toFixed(2).replace('.', ',')
            retorna '1545,50'
    tmb por localidade país
        n1.toLocaleString('pt-BR', {style: 'currency', currency: 'BRL'})
            retorna 'R$ 1,545.50'
